<script>
const STORAGE_KEY='goal_tracker_v1';

function load(){
  try{return JSON.parse(localStorage.getItem(STORAGE_KEY))||[];}catch(e){return[];}
}
function save(){localStorage.setItem(STORAGE_KEY,JSON.stringify(goals));}

let goals=load();
let nextId=goals.length?Math.max(...goals.map(g=>g.id))+1:1;

const RING_R=24,RING_C=28,CIRC=2*Math.PI*24;
function pct(g){
  if(g.type==='number') return Math.min(100,Math.round(g.current/g.target*100));
  return Math.min(100,Math.round(g.checked.filter(Boolean).length/g.days*100));
}
function ringColor(p){return p>=80?'#639922':p>=40?'#185FA5':'#BA7517';}
function renderRing(p){
  const dash=CIRC*(p/100),color=ringColor(p);
  return `<svg width="56" height="56" viewBox="0 0 56 56">
    <circle cx="28" cy="28" r="24" fill="none" stroke="#ebebeb" stroke-width="4"/>
    <circle cx="28" cy="28" r="24" fill="none" stroke="${color}" stroke-width="4" stroke-dasharray="${dash} ${CIRC}" stroke-linecap="round"/>
  </svg>`;
}
function streak(g){let s=0;for(let i=g.checked.length-1;i>=0;i--){if(g.checked[i])s++;else break;}return s;}

function renderNumber(g){
  const p=pct(g),color=ringColor(p);
  return `<div class="goal-card">
    <div class="card-header">
      <div><div class="card-title">${esc(g.name)}</div><span class="card-type type-num">数字型</span></div>
      <button class="del-btn" onclick="deleteGoal(${g.id})">×</button>
    </div>
    <div class="ring-row">
      <div class="ring-wrap">${renderRing(p)}<div class="ring-label">${p}%</div></div>
      <div class="meta">
        <div class="meta-main">${g.current} <span style="font-size:14px;font-weight:400;color:#888;">/ ${g.target} ${esc(g.unit)}</span></div>
        <div class="meta-sub">还剩 ${+(g.target-g.current).toFixed(2)} ${esc(g.unit)}</div>
      </div>
    </div>
    <div class="prog-bar-bg"><div class="prog-bar-fill" style="width:${p}%;background:${color};"></div></div>
    <div class="log-row">
      <input type="number" id="inp-${g.id}" min="0" placeholder="记录数量" />
      <button class="log-btn" onclick="logNum(${g.id})">+ 记录</button>
    </div>
  </div>`;
}

function renderHabit(g){
  const p=pct(g),color=ringColor(p);
  const done=g.checked.filter(Boolean).length;
  const nextUncheck=g.checked.findIndex(x=>!x);
  const dots=g.checked.map((d,i)=>`<div class="dot${d?' done':''}${i===nextUncheck?' today':''}" onclick="toggleDot(${g.id},${i})" title="第${i+1}天"></div>`).join('');
  return `<div class="goal-card">
    <div class="card-header">
      <div><div class="card-title">${esc(g.name)}</div><span class="card-type type-habit">习惯型</span></div>
      <button class="del-btn" onclick="deleteGoal(${g.id})">×</button>
    </div>
    <div class="ring-row">
      <div class="ring-wrap">${renderRing(p)}<div class="ring-label">${p}%</div></div>
      <div class="meta">
        <div class="meta-main">${done} <span style="font-size:14px;font-weight:400;color:#888;">/ ${g.days} 天</span></div>
        <div class="meta-sub">连续打卡 ${streak(g)} 天</div>
      </div>
    </div>
    <div class="prog-bar-bg"><div class="prog-bar-fill" style="width:${p}%;background:${color};"></div></div>
    <p style="font-size:12px;color:#888;margin-bottom:8px;">打卡记录</p>
    <div class="habit-dots">${dots}</div>
  </div>`;
}

function render(){
  const el=document.getElementById('goals-container');
  if(!goals.length){el.innerHTML='<div class="empty">还没有目标，点击右上角开始吧 🎯</div>';return;}
  const nums=goals.filter(g=>g.type==='number');
  const habits=goals.filter(g=>g.type==='habit');
  let html='';
  if(nums.length) html+=`<div class="section-title">数字目标</div><div class="goals-grid">${nums.map(renderNumber).join('')}</div>`;
  if(habits.length) html+=`<div class="section-title">习惯打卡</div><div class="goals-grid">${habits.map(renderHabit).join('')}</div>`;
  el.innerHTML=html;
}

function logNum(id){
  const inp=document.getElementById('inp-'+id);
  const val=parseFloat(inp.value);
  if(isNaN(val)||val<=0) return;
  const g=goals.find(x=>x.id===id);
  if(g){g.current=Math.min(g.target,+(g.current+val).toFixed(2));inp.value='';save();}
  render();
}
function toggleDot(id,i){
  const g=goals.find(x=>x.id===id);
  if(g){g.checked[i]=!g.checked[i];save();render();}
}
function deleteGoal(id){
  if(!confirm('确定删除这个目标？')) return;
  goals=goals.filter(x=>x.id!==id);save();render();
}
function openModal(){document.getElementById('modal').classList.add('open');setTimeout(()=>document.getElementById('m-name').focus(),50);}
function closeModal(){document.getElementById('modal').classList.remove('open');clearModal();}
function clearModal(){
  ['m-name','m-target','m-unit','m-days'].forEach(id=>{const el=document.getElementById(id);if(el)el.value='';});
  document.getElementById('m-type').value='number';toggleTypeFields();
}
function toggleTypeFields(){
  const t=document.getElementById('m-type').value;
  document.getElementById('m-num-fields').style.display=t==='number'?'':'none';
  document.getElementById('m-habit-fields').style.display=t==='habit'?'':'none';
}
function addGoal(){
  const name=document.getElementById('m-name').value.trim();
  const type=document.getElementById('m-type').value;
  if(!name) return alert('请输入目标名称');
  if(type==='number'){
    const target=parseFloat(document.getElementById('m-target').value);
    if(!target||target<=0) return alert('请输入有效的目标总量');
    const unit=document.getElementById('m-unit').value.trim()||'次';
    goals.push({id:nextId++,type:'number',name,target,current:0,unit});
  } else {
    const days=parseInt(document.getElementById('m-days').value)||30;
    goals.push({id:nextId++,type:'habit',name,days,checked:Array(days).fill(false)});
  }
  save();closeModal();render();
}
document.getElementById('modal').addEventListener('click',e=>{if(e.target===document.getElementById('modal'))closeModal();});
document.addEventListener('keydown',e=>{if(e.key==='Escape')closeModal();});
render();
</script>
